	 		//标签选择器

	 			$("li").toggleClass("cssClass");
	 		})
	 		//通配选择器
	 		$("#btn2").click(function(){
	 			$("*").toggleClass("cssClass");
	 		})
	 		//群组选择器
	 		$("#btn3").click(function(){
	 			$("ul,span").toggleClass("cssClass");
	 		})
	 		//后代选择器
	 		$("#btn4").click(function(){
	 			$("ul>li").toggleClass("cssClass");
	 		})
	 		//过滤选择器
	 		$("#btn5").click(function(){
	 			$("ul li:first").toggleClass("cssClass");
	 		})
	 		//属性选择器
	 		$("#btn6").click(function(){
	 			$("input[name=ip1]").toggleClass("cssClass");
	 		})
