web存储

				var storageObj = {
		 storageFun: function (){
			 if (!window.localStorage) {   //判断是否支持localStorage
				console.log("不支持");
				return false;
			} else{
				return true;
			}
		},
		Set: function (_item,value){//增加
			window.localStorage.setItem(_item,value);
		},
		Get: function (_item){//查找
			return window.localStorage.getItem(_item);
		},
		Clear:function(){//清除全部
			window.localStorage.clear();
		},
		Remove:function(){//删除单个
			window.localStorage.remove();
		},
		};
